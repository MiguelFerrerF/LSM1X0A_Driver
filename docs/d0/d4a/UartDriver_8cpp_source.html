89{

92
93 while (true) {

95 // Wait for an event from the UART queue
96 if (xQueueReceive(uart_event_queue, (void*)&event, (TickType_t)portMAX_DELAY)) {
97 // Clear temporary buffer
98 memset(dtmp, 0, RX_BUF_SIZE);
99 switch (event.type) {
100 // --- NORMAL CASE: DATA RECEIVED ---
101 case UART_DATA:
102 // Read exactly the number of bytes indicated by the event
103 uart_read_bytes(_uart_port, dtmp, event.size, portMAX_DELAY);
104
105 // Invoke callback passing the context
106 if (_receive_callback) {
107 _receive_callback(_callback_context, dtmp, event.size);
108 }
109 break;
110

114 uart_flush_input(_uart_port); // Clear to recover state

116 break;

119 LSM_LOG_ERROR("UART", "Error: UART Ring Buffer Full. Software too slow.");
120 uart_flush_input(_uart_port);
121 xQueueReset(uart_event_queue);
122 break;
123
124 case UART_BREAK:
125 // Sometimes useful to detect disconnections or resets
126 break;

129 case UART_FRAME_ERR:
