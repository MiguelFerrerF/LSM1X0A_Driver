8bool LSM1x0A_LoRaWAN::join(LsmJoinMode joinMode, uint32_t timeoutMs)
9{
10 if (!_controller)
11 return false;
12
13 char cmd[16];
14 snprintf(cmd, sizeof(cmd), "%s%d", LsmAtCommand::JOIN, joinMode);
15
16 // Save the desired join mode to use it in recovery if necessary
17 _joinMode = joinMode;
18
19 // Clear previous flags
20 _controller->clearEvents(LSM_EVT_JOIN_SUCCESS | LSM_EVT_JOIN_FAIL);
21
22 LSM_LOG_INFO("LORA", "Initiating LoRaWAN Join (%s)", joinMode == LsmJoinMode::OTAA ? "OTAA" : "ABP");
23
24 // Send the command and wait for the initial OK
25 AtError err = _controller->sendCommand(cmd, 2000, 1);
26 if (err != AtError::OK) {
27 LSM_LOG_ERROR("LORA", "Failed to dispatch Join request.");
28 return false;
29 }
30
31 LSM_LOG_INFO("LORA", "Join request dispatched successfully.");
32
33 // Now wait asynchronously for the URC event
34 uint32_t result = _controller->waitForEvent(LSM_EVT_JOIN_SUCCESS | LSM_EVT_JOIN_FAIL, timeoutMs);
35
36 if (result & LSM_EVT_JOIN_SUCCESS) {
37 if (_pendingChannelMask) {
38 setChannelMask(_cachedBand, _cachedSubBandMask);
39 }
40 vTaskDelay(pdMS_TO_TICKS(500));
41 LSM_LOG_INFO("LORA", "Join SUCCESS.");
42 return true; // Complete success
43 }
44 else if (result & LSM_EVT_JOIN_FAIL) {
45 LSM_LOG_ERROR("LORA", "Join FAILED (e.g. bad credentials or no coverage).");
46 return false; // Join failure (e.g., bad credentials, no coverage, etc.)
47 }
48
49 // If we reach here, it's an unexpected MAC layer timeout
50 LSM_LOG_ERROR("LORA", "Join TIMEOUT. Module hung without emitting failure event. Forcing recovery.");
51 _controller->recoverModule();
52 return false; // Timeout
53}

55bool LSM1x0A_LoRaWAN::sendData(uint8_t port, const char* data, bool confirmed, uint32_t timeoutMs)
56{
57 if (!_controller)
58 return false;
59 if (!isJoined())
60 return false;
61 if (port < 1 || port > 223)
62 return false;
63
64 if (data == nullptr)
65 return false;
66
67 // AT+SEND=<Port>:<Ack>:<Payload>
68 char cmd[256];
69 snprintf(cmd, sizeof(cmd), "AT+SEND=%d:%d:%s", port, confirmed ? 1 : 0, data);
70
71 // Clear previous flags
72 _controller->clearEvents(LSM_EVT_TX_SUCCESS | LSM_EVT_TX_FAIL | LSM_EVT_RX_TIMEOUT);
73
74 LSM_LOG_INFO("LORA", "Sending LoRaWAN package (Port: %d, Confirmed: %d)", port, confirmed);
75 LSM_LOG_VERBOSE("LORA", "Payload: %s", data);
76
77 // Send the payload (this responds OK quickly if well-formatted and not busy)
78 AtError err = _controller->sendCommand(cmd, 3000, 1);
79 if (err != AtError::OK) {
80 LSM_LOG_ERROR("LORA", "Failed to dispatch LoRaWAN transmission request.");
81 return false;
82 }
83
84 // If timeoutMs is 0 (Auto), calculate dynamically based on retries
85 uint32_t calcTimeoutMs = timeoutMs;
86
87 if (confirmed) {
88 int retries = -1;
89 if (_cachedConfirmRetry >= 0)
90 retries = _cachedConfirmRetry;
91 else
92 retries = getConfirmRetry();
93 if (retries <= 0)
94 retries = 3; // Fallback
95
96 // Confirmed Retries = 1 initial + N retries. So total transmits = retries + 1.
97 // Each transmit yields 2 RX_TIMEOUTs if no ACK is received.
98 int expectedTimeouts = (retries + 1) * 2;
99 int receivedTimeouts = 0;
100
101 // Base safety 5s + each attempt takes ~4.5 seconds of radio (TX + Rx1 + Rx2 + MAC delay)
102 uint32_t dynamicTimeoutMs = 5000 + ((retries + 1) * 4500);
103 if (calcTimeoutMs == 0 || calcTimeoutMs < dynamicTimeoutMs) {
104 calcTimeoutMs = dynamicTimeoutMs;
105 }
106
107 uint32_t startMs = (uint32_t)(esp_timer_get_time() / 1000);
108 _controller->clearEvents(LSM_EVT_RX_TIMEOUT);
109
110 while (((uint32_t)(esp_timer_get_time() / 1000) - startMs) < calcTimeoutMs) {
111 uint32_t expectedBits = LSM_EVT_TX_SUCCESS | LSM_EVT_TX_FAIL | LSM_EVT_RX_TIMEOUT;
112 if (_linkCheckRequested)
113 expectedBits |= LSM_EVT_LINK_CHECK_ANS;
114
115 // Wait iteratively for events
116 uint32_t result = _controller->waitForEvent(expectedBits, 3000, true);
117
118 if (result & LSM_EVT_LINK_CHECK_ANS) {
119 _linkCheckRequested = false;
120 return true;
121 }
122 if (result & LSM_EVT_TX_SUCCESS) {
123 if (!_linkCheckRequested)
124 return true; // Downlink intercepted or quick success
125 }
126 if (result & LSM_EVT_TX_FAIL) {
127 _linkCheckRequested = false;
128 return false;
129 }
130 if (result & LSM_EVT_RX_TIMEOUT) {
131 receivedTimeouts++;
132 if (receivedTimeouts >= expectedTimeouts) {
133 if (_linkCheckRequested) {
134 _linkCheckRequested = false;
135 }
136 return false; // MAC layer exhausted retries gracefully. NO recoverModule().
137 }
138 }
139 }
140
141 // If we reach a total asynchronous timeout, it means the module hung (no output)
142 LSM_LOG_ERROR("LORA", "MAC transmission timeout: Module hung without outputting result.");
143 _controller->recoverModule();
144 return false; // Real MAC layer error (no UART output)
145 }
146 else {
147 // For pure Unconfirmed, we wait for `MAC rxTimeOut` (2 per transmission attempt)
148 int retries = -1;
149 if (_cachedUnconfirmRetry >= 0)
150 retries = _cachedUnconfirmRetry;
151 else
152 retries = getUnconfirmRetry();
153 if (retries < 0)
154 retries = 5; // Fallback if error occurred
155
156 // (N retries) * 2 RX windows
157 int expectedTimeouts = retries * 2;
158 int receivedTimeouts = 0;
159
160 // As retries can take time, extend the base timeout for dynamic safety
161 uint32_t dynamicTimeoutMs = 5000 + (retries * 3500);
162 if (calcTimeoutMs == 0 || calcTimeoutMs < dynamicTimeoutMs) {
163 calcTimeoutMs = dynamicTimeoutMs;
164 }
165
166 uint32_t startMs = (uint32_t)(esp_timer_get_time() / 1000);
167 _controller->clearEvents(LSM_EVT_RX_TIMEOUT); // clear before starting
168
169 while (((uint32_t)(esp_timer_get_time() / 1000) - startMs) < calcTimeoutMs) {
170 uint32_t expectedBits = LSM_EVT_TX_SUCCESS | LSM_EVT_RX_TIMEOUT;
171 if (_linkCheckRequested)
172 expectedBits |= LSM_EVT_LINK_CHECK_ANS;
173
174 // Wait for a MAC RX timeout, or if magically a SUCCESS arrives (e.g., LinkCheck piggyback)
175 uint32_t result = _controller->waitForEvent(expectedBits, 3000, true);
176
177 if (result & LSM_EVT_LINK_CHECK_ANS) {
178 _linkCheckRequested = false;
179 return true; // Total success: metadata received
180 }
181
182 if (result & LSM_EVT_TX_SUCCESS) {
183 if (!_linkCheckRequested)
184 return true; // Downlink intercepted or MAC finished early
185 }
186
187 if (result & LSM_EVT_RX_TIMEOUT) {
188 receivedTimeouts++;
189 if (receivedTimeouts >= expectedTimeouts) {
190 if (_linkCheckRequested) {
191 _linkCheckRequested = false;
192 _controller->recoverModule();
193 return false;
194 }
195 return true;
196 }
197 }
198 }
199
200 // If we reach a total asynchronous timeout, it means the module hung (no output)
201 LSM_LOG_ERROR("LORA", "MAC transmission timeout: Module hung without outputting result.");
202 _controller->recoverModule();
203 return false; // Global wait timeout
204 }

207bool LSM1x0A_LoRaWAN::requestLinkCheck()
208{
209 if (!_controller)
210 return false;
211 if (!isJoined())
212 return false;
213
214 // AT+LINKC does not generate an immediate event, it only schedules the request for the next Uplink
215 LSM_LOG_INFO("LORA", "Requesting LinkCheck for the next uplink.");
216 AtError err = _controller->sendCommand(LsmAtCommand::LINK_CHECK, 2000, 1);
217 if (err == AtError::OK) {
218 _linkCheckRequested = true;
219 return true;
220 }
221 return false;
222}

234bool LSM1x0A_LoRaWAN::recoverConnection(int maxRetries)
235{
236 for (int i = 0; i < maxRetries; i++) {
237 // Here we assume OTAA by default, although in the future the mode could be cached
238 if (join(_joinMode)) {
239 return true;
240 }
241 vTaskDelay(pdMS_TO_TICKS(2000)); // Pause between attempts
242 }
243 return false;
