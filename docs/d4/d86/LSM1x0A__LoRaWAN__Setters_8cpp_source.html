5// Helper function to format hex strings with colons (e.g. "0011223344556677" or "00:11:22:33:44:55:66:77" -> "00:11:22:33:44:55:66:77")
6static bool formatHexWithColons(const char* in, char* out, size_t outSize, size_t expectedBytes)
7{
8 if (!in || !out) return false;
9
10 char cleanHex[64]; // max 32 bytes * 2 = 64
11 size_t cleanIdx = 0;
12
13 for (size_t i = 0; in[i] != '\0'; ++i) {
14 if (in[i] == ':') continue;
15
16 // Check if valid hex char
17 if ((in[i] >= '0' && in[i] <= '9') ||
18 (in[i] >= 'a' && in[i] <= 'f') ||
19 (in[i] >= 'A' && in[i] <= 'F')) {
20 if (cleanIdx < sizeof(cleanHex)) {
21 cleanHex[cleanIdx++] = in[i];
22 } else {
23 return false; // Too large
24 }
25 } else {
26 return false; // Invalid hex character
27 }
28 }
29
30 // Check if the total hex characters match the expected length
31 if (cleanIdx != expectedBytes * 2) {
32 return false;
33 }
34
35 size_t outIdx = 0;
36 for (size_t i = 0; i < expectedBytes; ++i) {
37 if (outIdx + 3 > outSize) {
38 return false;
39 }
40 out[outIdx++] = cleanHex[i * 2];
41 out[outIdx++] = cleanHex[i * 2 + 1];
42 if (i < expectedBytes - 1) {
43 out[outIdx++] = ':';
44 }
45 }
46 out[outIdx] = '\0';
47 return true;
48}

322bool LSM1x0A_LoRaWAN::setChannelMask(LsmBand band, uint16_t subBandMask)
323{
324 _cachedBand = band;
325 _cachedSubBandMask = subBandMask;
326
327 if (!isJoined()) {
328 _pendingChannelMask = true;
329 return true;
330 }
331
332 _pendingChannelMask = false;
333 int maxCount = 0;
334
335 switch (band) {
336 case LsmBand::EU868:
337 case LsmBand::AS923_1:
338 case LsmBand::KR920:
339 case LsmBand::IN865:
340 case LsmBand::RU864:
341 case LsmBand::AS923_4:
342 case LsmBand::EU433:
343 case LsmBand::CN779:
344 maxCount = 1;
345 break;
346 case LsmBand::US915:
347 case LsmBand::AU915:
348 case LsmBand::CN470:
349 maxCount = 6;
350 break;
351 default:
352 return false; // Unknown Band length
353 }
354
355 uint16_t blocks[6] = {0};
356
357 if (subBandMask == 0xFFFF) {
358 if (maxCount == 1) {
359 blocks[0] = 0x00FF;
360 }
361 else if (maxCount == 6) {
362 blocks[0] = 0xFFFF;
363 blocks[1] = 0xFFFF;
364 blocks[2] = 0xFFFF;
365 blocks[3] = 0xFFFF;
366 if (band == LsmBand::CN470) {
367 blocks[4] = 0xFFFF;
368 blocks[5] = 0xFFFF;
369 }
370 else {
371 blocks[4] = 0x00FF; // US915 has 8 500kHz channels
372 blocks[5] = 0x0000;
373 }
374 }
375 }
376 else {
377 if (maxCount == 1) {
378 uint16_t val = 0;
379 if (subBandMask & 0x0001)
380 val |= 0x000F;
381 if (subBandMask & 0x0002)
382 val |= 0x00F0;
383 blocks[0] = val;
384 }
385 else {
386 int maxSubBands = (band == LsmBand::CN470) ? 12 : 8;
387 for (int i = 0; i < maxSubBands; i++) {
388 if (subBandMask & (1 << i)) {
389 // For subband i (0 to 7 usually):
390 // 125kHz channels
391 int blockIdx = i / 2;
392 uint16_t offsetHex = ((i % 2) == 0) ? 0x00FF : 0xFF00;
393 blocks[blockIdx] |= offsetHex;
394
395 // 500kHz channels (for US915/AU915 only)
396 if (band == LsmBand::US915 || band == LsmBand::AU915) {
397 blocks[4] |= (1 << i);
398 }
399 }
400 }
401 }
402 }
403
404 char maskStr[128] = {0};
405 size_t currentLen = 0;
406 for (int i = 0; i < maxCount; i++) {
407 int written = 0;
408 if (i > 0) {
409 written = snprintf(maskStr + currentLen, sizeof(maskStr) - currentLen, ":%04X", blocks[i]);
410 }
411 else {
412 written = snprintf(maskStr + currentLen, sizeof(maskStr) - currentLen, "%04X", blocks[i]);
413 }
414 if (written > 0 && currentLen + written < sizeof(maskStr)) {
415 currentLen += written;
416 }
417 }
418
419 if (strlen(maskStr) == 0)
420 return false;
421
422 char cmd[128];
423 snprintf(cmd, sizeof(cmd), "%s%s", LsmAtCommand::CHANNEL_MASK, maskStr);
424 return _controller->sendCommand(cmd, 1000) == AtError::OK;
425}

532bool LSM1x0A_LoRaWAN::restoreConfig()
533{
534 bool success = true;
535
536 // Restore identifiers if they were cached
537 if (_cachedDevEui[0] != '\0')
538 if (!setDevEUI(_cachedDevEui))
539 success = false;
540 if (_cachedDevAddr[0] != '\0')
541 if (!setDevAddr(_cachedDevAddr))
542 success = false;
543 if (_cachedNwkID[0] != '\0')
544 if (!setNwkID((int)strtol(_cachedNwkID, NULL, 16)))
545 success = false;
546
547 // Restore Band and SubBand if possible
548 if (_cachedBand != LsmBand::BAND_UNKNOWN) {
549 if (!setBand(_cachedBand))
550 success = false;
551 if (_cachedSubBandMask != 0xFFFF && _cachedBand == LsmBand::US915)
552 if (!setChannelMask(_cachedBand, _cachedSubBandMask))
553 success = false;
554 }
555
556 // Restore network configurations
557 if (_cachedAdrEnabled != -1)
558 if (!setADR(_cachedAdrEnabled == 1))
559 success = false;
560 if (_cachedDataRate != LsmDataRate::DR_UNKNOWN)
561 if (!setDataRate(_cachedDataRate))
562 success = false;
563 if (_cachedTxPower != LsmTxPower::TP_UNKNOWN)
564 if (!setTxPower(_cachedTxPower))
565 success = false;
566 if (_cachedDutyCycle != -1)
567 if (!setDutyCycle(_cachedDutyCycle == 1))
568 success = false;
569
570 // Restore delays
571 if (_cachedJoin1Delay != -1)
572 if (!setJoin1Delay(_cachedJoin1Delay))
573 success = false;
574 if (_cachedJoin2Delay != -1)
575 if (!setJoin2Delay(_cachedJoin2Delay))
576 success = false;
577 if (_cachedRx1Delay != -1)
578 if (!setRx1Delay(_cachedRx1Delay))
579 success = false;
580 if (_cachedRx2Delay != -1)
581 if (!setRx2Delay(_cachedRx2Delay))
582 success = false;
583
584 // Restore RX2 frequencies / data rates
585 if (_cachedRx2DataRate != LsmDataRate::DR_UNKNOWN)
586 if (!setRx2DataRate(_cachedRx2DataRate))
587 success = false;
588 if (_cachedRx2Frequency > 0)
589 if (!setRx2Frequency(_cachedRx2Frequency))
590 success = false;
591
592 // Restore retries
593 if (_cachedConfirmRetry >= 0)
594 if (!setConfirmRetry(_cachedConfirmRetry))
595 success = false;
596 if (_cachedUnconfirmRetry >= 0)
597 if (!setUnconfirmRetry(_cachedUnconfirmRetry))
598 success = false;
599
600 return success;
601}

603bool LSM1x0A_LoRaWAN::loadConfigFromModule()
604{
605 bool success = true;
606
607 // Call internal getters to force their values to be overwritten in the current cache
608 if (!getDevEUI(_cachedDevEui, sizeof(_cachedDevEui)))
609 success = false;
610 if (!getDevAddr(_cachedDevAddr, sizeof(_cachedDevAddr)))
611 success = false;
612
613 int nwkId = getNwkID();
614 if (nwkId >= 0) {
615 snprintf(_cachedNwkID, sizeof(_cachedNwkID), "%08X", (unsigned int)nwkId);
616 }
617 else {
618 success = false;
619 }
620
621 if (getADR() < 0)
622 success = false;
623 if (getDataRate() == LsmDataRate::DR_UNKNOWN)
624 success = false;
625 if (getTxPower() == LsmTxPower::TP_UNKNOWN)
626 success = false;
627 if (getBand() == LsmBand::BAND_UNKNOWN)
628 success = false;
629 if (getDutyCycle() < 0)
630 success = false;
631
632 if (getJoin1Delay() < 0)
633 success = false;
634 if (getJoin2Delay() < 0)
635 success = false;
636 if (getRx1Delay() < 0)
637 success = false;
638 if (getRx2Delay() < 0)
639 success = false;
640
641 if (getRx2DataRate() == LsmDataRate::DR_UNKNOWN)

643 if (getRx2Frequency() < 0)
644 success = false;
