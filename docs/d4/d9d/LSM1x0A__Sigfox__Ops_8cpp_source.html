54bool LSM1x0A_Sigfox::sendBit(bool bit, bool downlink, uint8_t txRepeat)
55{
56 char cmd[32];
57 if (downlink || txRepeat != 2) {
58 snprintf(cmd, sizeof(cmd), "%s%d,%d,%d", LsmAtCommand::SEND_BIT, bit ? 1 : 0, downlink ? 1 : 0, txRepeat);
59 }
60 else {
61 snprintf(cmd, sizeof(cmd), "%s%d", LsmAtCommand::SEND_BIT, bit ? 1 : 0);
62 }
63
64 if (downlink) {
65 char rxBuffer[32] = {0};
66 LSM_LOG_INFO("SIGFOX", "Sending Bit (%d) w/ Downlink requested", bit);
67 AtError err = _controller->sendCommandWithResponse(cmd, rxBuffer, sizeof(rxBuffer), "+RX_H=", 60000);
68 if (err == AtError::OK) {
69 parseSigfoxDownlink(rxBuffer);
70 LSM_LOG_INFO("SIGFOX", "Bit TX Success - Downlink received and parsed.");
71 return true;
72 }
73 // In Sigfox, the downlink is often lost more easily, or the timeout occurs.
74 // We return false to indicate that there was NO downlink, but the network did consume the OOB/Uplink request.
75 LSM_LOG_WARN("SIGFOX", "Bit TX executed but Downlink was not successfully received.");
76 return false;
77 }
78
79 // Use a longer timeout (30s) because Sigfox TX window takes some time
80 LSM_LOG_INFO("SIGFOX", "Sending Bit (%d) without Downlink", bit);
81 bool success = _controller->sendCommand(cmd, 30000) == AtError::OK;
82 if (!success) {
83 LSM_LOG_ERROR("SIGFOX", "Failed to send Bit.");
84 }
85 return success;
86}

168bool LSM1x0A_Sigfox::parseSigfoxDownlink(const char* rxBuffer)
169{
170 if (!rxBuffer || strlen(rxBuffer) < 12)
171 return false;
172
173 char hexTemp[9];
174 // Parse TIMESTAMP (First 8 characters)
175 memcpy(hexTemp, rxBuffer, 8);
176 hexTemp[8] = '\0';
177 uint32_t ts = strtoul(hexTemp, NULL, 16);
178
179 // Adjust the ESP32 internal clock with this time
180 time_t rawTime = (time_t)ts;
181 struct timeval tv = {.tv_sec = static_cast<long>(rawTime), .tv_usec = 0};
182 settimeofday(&tv, NULL);
183
184 _lastDownlinkTime = rawTime;
185
186 // Parse RSSI (Next 4 characters)
187 char rssiTemp[5];
188 memcpy(rssiTemp, rxBuffer + 8, 4);
189 rssiTemp[4] = '\0';
190 uint32_t rawRssi = strtoul(rssiTemp, NULL, 16);
191 _lastRxRSSI = (int16_t)rawRssi;
192
193 // Log the parsed downlink information
194 struct tm* timeInfo;
195 timeInfo = gmtime(&rawTime);
196
197 LSM_LOG_VERBOSE("SIGFOX", "Parsed Downlink - TS: %04d-%02d-%02d %02d:%02d:%02d, RSSI: %d", timeInfo->tm_year + 1900, timeInfo->tm_mon + 1,
198 timeInfo->tm_mday, timeInfo->tm_hour, timeInfo->tm_min, timeInfo->tm_sec, _lastRxRSSI);
199
200 return true;
201}

274bool LSM1x0A_Sigfox::isValidHex(const char* str, size_t maxLen)
275{
276 if (str == nullptr) {
277 return false;
278 }
279
280 size_t length = 0;
281 while (*str && length < maxLen) {
282 if (!((*str >= '0' && *str <= '9') || (*str >= 'A' && *str <= 'F') || (*str >= 'a' && *str <= 'f'))) {
283 return false; // Non-hexadecimal character detected
284 }
285 str++;
286 length++;
287 }
288
289 if (length >= maxLen && *str != '\0') {
290 return false;
291 }
292 return true;
