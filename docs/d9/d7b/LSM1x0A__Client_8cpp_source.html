137bool LSM1x0A_Client::sendPayload(const uint8_t* payload, size_t length, bool requestAck, uint8_t port, bool enableRetries, uint8_t maxRetries)
138{
139 if (!payload || length == 0)
140 return false;
141
142 uint8_t attempts = enableRetries ? (maxRetries + 1) : 1;
143 bool success = false;
144
145 if (_configuredMode == LsmMode::LORAWAN) {
146 // Convert to hex string
147 size_t safeLength = length;
148 if (safeLength > 242)
149 safeLength = 242; // Prevent overflow on 512 byte buffer (max LoRaWAN payload is 242 anyways)
150 char hexString[512];
151 for (size_t j = 0; j < safeLength; j++) {
152 sprintf(&hexString[j * 2], "%02X", payload[j]);
153 }
154 hexString[safeLength * 2] = '\0';
155
156 for (uint8_t i = 0; i < attempts; i++) {
157 success = _controller->lorawan.sendData(port, hexString, requestAck);
158 if (success)
159 return true;
160
161 if (requestAck && enableRetries && i == attempts - 1) {
162 _controller->recoverModule();
163 }
164 else if (!success && i < attempts - 1) {
165 vTaskDelay(pdMS_TO_TICKS(1000));
166 }
167 }
168 }
169 else if (_configuredMode == LsmMode::SIGFOX) {
170 for (uint8_t i = 0; i < attempts; i++) {
171 success = _controller->sigfox.sendPayload(payload, length, requestAck, 2);
172 if (success)
173 return true;
174 if (!success && i < attempts - 1) {
175 vTaskDelay(pdMS_TO_TICKS(1000));
176 }
177 }
178 }
179
180 return false;
181}

267bool LSM1x0A_Client::_charsToBytes(const char* hexString, uint8_t* outputBuffer, size_t maxLen)
268{
269 if (!hexString || !outputBuffer)
270 return false;
271 size_t hexLen = strlen(hexString);
272 if (hexLen % 2 != 0 || (hexLen / 2) > maxLen)
273 return false;
274
275 for (size_t i = 0; i < hexLen / 2; i++) {
276#ifdef _MSC_VER
277 sscanf_s(&hexString[i * 2], "%2hhx", &outputBuffer[i]);
278#else
279 sscanf(&hexString[i * 2], "%2hhx", &outputBuffer[i]);
280#endif
281 }
282 return true;

296void LSM1x0A_Client::_onRxData(void* ctx, const char* payload)
297{
298 LSM1x0A_Client* self = static_cast<LSM1x0A_Client*>(ctx);
299 if (!self || !self->_downlinkCallback || !payload)
300 return;
301
302 char temp[256];
303 strncpy(temp, payload, sizeof(temp) - 1);
304 temp[sizeof(temp) - 1] = '\0';
305
306 char* portStr = strtok(temp, ":");
307 char* sizeStr = strtok(NULL, ":");
308 char* dataStr = strtok(NULL, ":");
309
310 if (!portStr || !sizeStr || !dataStr)
311 return;
312
313 uint8_t port = (uint8_t)strtol(portStr, NULL, 10);
314 size_t size = (size_t)strtol(sizeStr, NULL, 16);
315
316 uint8_t binPayload[128];
317 if (size > sizeof(binPayload))
318 size = sizeof(binPayload);
319
320 if (self->_charsToBytes(dataStr, binPayload, size)) {
321 self->_downlinkCallback(port, binPayload, size);
322 }
