72void LSM1x0A_AtParser::eatBuffer(const uint8_t* data, size_t len)
73{
74 for (size_t i = 0; i < len; i++) {
75 char c = (char)data[i];
76 // Standard end-of-line detection \r\n
77 if (c == '\n' || c == '\r') {
78 if (_lineIdx > 0) {
79 _lineBuffer[_lineIdx] = '\0'; // Null-terminate
80 processLine(_lineBuffer);
81 _lineIdx = 0; // Reset for next line
82 }
83 }
84 else {
85 // Protection against internal buffer overflow
86 if (_lineIdx < AT_BUFFER_SIZE - 1) {
87 _lineBuffer[_lineIdx++] = c;
88 }
89 // If full, ignore the rest of the line until \n
90 }
91 }

97void LSM1x0A_AtParser::processLine(char* line)
98{
99 // 1. LIMPIEZA
100 size_t len = strlen(line);
101 while (len > 0 && (line[len - 1] == '\r' || line[len - 1] == '\n'))
102 line[--len] = '\0';
103 if (len == 0)
104 return;
105
106 LSM_LOG_VERBOSE("UART", "RX Line: %s", line);
107
108 // 2. DETECTION OF ASYNCHRONOUS EVENTS (+EVT)
109 // Based on the functions: AT_event_join, AT_event_receive, AT_event_confirm
110 char* evtPtr = strstr(line, "+EVT:");
111
112 if (evtPtr != NULL) {
113 char* payload = evtPtr + 5; // Skip the "+EVT:" prefix
114 if (_eventCallback) {
115 // --- A. JOIN ---
116 if (strncmp(payload, "JOINED", 6) == 0) {
117 _eventCallback(LsmEvent::JOIN, "SUCCESS", _eventCtx);
118 }
119 else if (strncmp(payload, "JOIN FAILED", 11) == 0) {
120 _eventCallback(LsmEvent::JOIN, "FAILED", _eventCtx);
121 }
122
123 // --- B. DATA RECEPTION (RECV) ---
124 // Format: RECV_CONFIRMED:PORT:SIZE:DATA...
125 else if (strncmp(payload, "RECV_", 5) == 0) {
126 // Search for the first ':' to skip CONFIRMED/UNCONFIRMED
127 char* ptr = strchr(payload, ':');
128 if (ptr) {
129 // Pass "PORT:SIZE:DATA" clean to the user
130 _eventCallback(LsmEvent::RX_DATA, ptr + 1, _eventCtx);
131 }
132 }
133
134 // --- C. RECEPTION METADATA (RX_) ---
135 // Format: RX_1, PORT 2, DR 5, RSSI -90, SNR 10...
136 // NOTE: Do not confuse with RECV_. We check RX_
137 else if (strncmp(payload, "RX_", 3) == 0) {
138 // Pass the entire metadata string for the helper to parse
139 _eventCallback(LsmEvent::RX_META, payload, _eventCtx);
140 }
141
142 // --- D. SEND CONFIRMATION (TX) ---
143 else if (strncmp(payload, "SEND_CONFIRMED", 14) == 0) {
144 _eventCallback(LsmEvent::TX, "SUCCESS", _eventCtx);
145 }
146 else if (strncmp(payload, "SEND_FAILED", 11) == 0) {
147 _eventCallback(LsmEvent::TX, "FAILED", _eventCtx);
148 }
149
150 // --- E. CLASS CHANGE ---
151 else if (strncmp(payload, "SWITCH_TO_CLASS_", 16) == 0) {
152 // Payload will be "SWITCH_TO_CLASS_C", we pass "C"
153 _eventCallback(LsmEvent::CLASS, payload + 16, _eventCtx);
154 }
155
156 // --- F. BEACONS ---
157 else if (strncmp(payload, "BEACON_LOST", 11) == 0) {
158 _eventCallback(LsmEvent::BEACON, "LOST", _eventCtx);
159 }
160 else if (strncmp(payload, "RX_BC", 5) == 0) {
161 // Beacon received with info
162 _eventCallback(LsmEvent::BEACON, payload, _eventCtx);
163 }
164 }
165 return; // Event processed, do not continue searching for command responses
166 }
167
168 // 3. SPECIAL CASES WITHOUT +EVT PREFIX
169 if (strncmp(line, "NVM DATA", 8) == 0) {
170 if (_eventCallback)
171 _eventCallback(LsmEvent::NVM, line, _eventCtx);
172 return;
173 }
174
175 if (strncmp(line, "channel_mask[", 13) == 0) {
176 if (_eventCallback)
177 _eventCallback(LsmEvent::CHMASK, line, _eventCtx);
178 // IMPORTANT: Do not return; if it is a synchronous command waiting for the final 'OK'.
179 // This will send the asynchronous event and then the Parser will continue running waiting for the OK.
180 }
181
182 // 4. Ignore irrelevant lines
183 // Discard asynchronous echoes of the command itself (e.g., "AT+BAT=?")
184 if (strncmp(line, "AT+", 3) == 0)
185 return;
186
187 // If we reach here, the line did NOT have +EVT, so if it starts with "confirmed flag",
188 // it is isolated garbage and we discard it.
189 if (strncmp(line, "confirmed flag:", 15) == 0)
190 return;
191 if (strncmp(line, "---", 3) == 0)
192 return; // Boot logs
193 if (strncmp(line, "+EVT:Prepare Frame", 18) == 0)
194 return;
195
196 // Device type detection
197 if (strstr(line, "LSM100A") != nullptr)
198 _deviceType = LsmModuleType::LSM100A;
199 else if (strstr(line, "LSM110A") != nullptr)
200 _deviceType = LsmModuleType::LSM110A;
201
202 // Operating mode detection
203 if (strstr(line, "LoRa") != nullptr)
204 _detectedMode = LsmMode::LORAWAN;
205 else if (strstr(line, "Sigfox") != nullptr)
206 _detectedMode = LsmMode::SIGFOX;
207
208 // Intercept MAC rxTimeOut explicitly without failing the pendingCommand
209 if (strstr(line, "MAC rxTimeOut") != nullptr) {
210 if (_eventCallback)
211 _eventCallback(LsmEvent::RX_TIMEOUT, "TIMEOUT", _eventCtx);
212 }
213
214 if (_pendingCommand) {
215 AtError err = parseErrorString(line);
216
217 // If the line is a final return code (OK, ERROR, etc... we finish)
218 if (err != AtError::UNKNOWN) {
219 _lastResultError = err;
220 xSemaphoreGive(_syncSem);
221 return;
222 }
223
224 // Data capture (Getters). Only if the line was NOT an OK, Error, or Echo
225 if (_userOutBuffer != nullptr) {
226 const char* dataStart = line;
227 bool match = (_expectedTag == nullptr); // If no tag, take the whole line
228
229 if (_expectedTag && strstr(line, _expectedTag)) {
230 dataStart = strstr(line, _expectedTag) + strlen(_expectedTag);
231 match = true;
232 }
233
234 if (match) {
235 while (*dataStart == ' ' || *dataStart == ':')
236 dataStart++; // Clean prefixes
237 strncpy(_userOutBuffer, dataStart, _userOutSize - 1);
238 _userOutBuffer[_userOutSize - 1] = '\0';
239 }
240 }
241 }
242}

247bool LSM1x0A_AtParser::parseRxMetadata(const char* payload, LsmRxMetadata* out)
248{
249 // Example: "RX_1, PORT 2, DR 5, RSSI -90, SNR 10, DMODM 10, GWN 2"
250 if (!payload || !out)
251 return false;
252
253 // Initialize defaults
254 memset(out, 0, sizeof(LsmRxMetadata));
255 out->hasLinkCheck = false;
256
257 // 1. Extract Slot (RX_1, RX_C, etc)
258 // The payload starts with "RX_"
259 const char* pStart = payload + 3; // Skip "RX_"
260 const char* pComma = strchr(pStart, ',');
261 if (pComma) {
262 int len = pComma - pStart;
263 if (len < sizeof(out->slot)) {
264 strncpy(out->slot, pStart, len);
265 out->slot[len] = '\0';
266 }
267 }
268 else
269 return false; // Invalid format
270
271 // 2. Extract standard numeric fields
272 // Look for key substrings
273 const char* pPort = strstr(payload, "PORT ");
274 const char* pDr = strstr(payload, "DR ");
275 const char* pRssi = strstr(payload, "RSSI ");
276 const char* pSnr = strstr(payload, "SNR ");
277
278 if (pPort)
279 out->port = atoi(pPort + 5);
280 if (pDr)
281 out->dataRate = atoi(pDr + 3);
282 if (pRssi)
283 out->rssi = atoi(pRssi + 5);
284 if (pSnr)
285 out->snr = atoi(pSnr + 4);
286
287 // 3. Extract LinkCheck
288 const char* pDmodm = strstr(payload, "DMODM ");
289 const char* pGwn = strstr(payload, "GWN ");
290
291 if (pDmodm && pGwn) {
292 out->hasLinkCheck = true;
293 out->demodMargin = atoi(pDmodm + 6);
294 out->nbGateways = atoi(pGwn + 4);
295 }
296
297 return true;
298}

338AtError LSM1x0A_AtParser::sendCommandWithResponse(const char* cmd, const char* expectedTag, char* outBuffer, size_t outSize, uint32_t timeoutMs)
339{
340 // 1. Ensure no other transaction is in progress
341 xSemaphoreTake(_syncSem, 0);
342
343 if (_pendingCommand) {
344 return AtError::BUSY;
345 }
346
347 // 2. Configure transaction context
348 _pendingCommand = true;
349 _lastResultError = AtError::TIMEOUT;
350 _expectedTag = expectedTag;
351 _userOutBuffer = outBuffer; // Pointer to user's stack/global buffer
352 _userOutSize = outSize;
353
354 // Clear user buffer for safety
355 if (outBuffer)
356 outBuffer[0] = '\0';
357
358 LSM_LOG_VERBOSE("UART", "TX: %s", cmd);
359
360 // 3. Send Command via UART
361 _driver->sendData(cmd, strlen(cmd));
362 _driver->sendData("\r\n", 2);
363
364 // 4. Block waiting for response
365 if (xSemaphoreTake(_syncSem, pdMS_TO_TICKS(timeoutMs)) != pdTRUE) {
366 LSM_LOG_DEBUG("UART", "sendCommandWithResponse timeout");
367 _pendingCommand = false;
368 return AtError::TIMEOUT;
369 }
370
371 _pendingCommand = false;
372 _userOutBuffer = nullptr; // Unlink buffer for safety
373
374 return _lastResultError;
375}
